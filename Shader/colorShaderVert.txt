#version 130

in vec2 vertexPosition;
in vec4 vertexColor;
in vec2 vertexUV;

out vec4 fragmentPosition;
out vec4 fragmentColor;
out vec2 fragmentUV;


void main(){
    gl_Position.xy = vertexPosition.xy;
    gl_Position.z = 0.0;
    gl_Position.w = 1.0;

    fragmentColor = vertexColor;
    fragmentPosition = gl_Position;
    fragmentUV = vec2(vertexUV.x, 1.0-vertexUV.y);
}
